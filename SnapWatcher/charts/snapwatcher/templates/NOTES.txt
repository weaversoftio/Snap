{{- if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ include "snapwatcher.namespace" . }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "snapwatcher.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ include "snapwatcher.namespace" . }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ include "snapwatcher.namespace" . }} svc -w {{ include "snapwatcher.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ include "snapwatcher.namespace" . }} {{ include "snapwatcher.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ include "snapwatcher.namespace" . }} -l "app.kubernetes.io/name={{ include "snapwatcher.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ include "snapwatcher.namespace" . }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ include "snapwatcher.namespace" . }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

To check the status of the deployment:
  kubectl get deployment {{ include "snapwatcher.fullname" . }} -n {{ include "snapwatcher.namespace" . }}

To view the logs:
  kubectl logs -f deployment/{{ include "snapwatcher.fullname" . }} -n {{ include "snapwatcher.namespace" . }}

To check if pods are being monitored:
  kubectl get pods -l snap.weaversoft.io/snap=true --all-namespaces

SnapWatcher will automatically checkpoint pods with the following labels:
  - snap.weaversoft.io/snap: "true"
  - snap.weaversoft.io/mutated: "false"
